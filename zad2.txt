 LJMP START
 ORG 100H
START:
 CALL LCD_CLR
 CALL WAIT_KEY
 
 MOV R1, A
 MOV B, #10
 DIV AB
 SWAP A
 ADD A, B
 CALL WRITE_HEX
 
 MOV A, #'-'
 CALL WRITE_DATA
 
 CALL WAIT_KEY
 
 MOV R2, A
 MOV B, #10
 DIV AB
 SWAP A
 ADD A, B
 CALL WRITE_HEX
 
 MOV A, #'='
 CALL WRITE_DATA
  
 MOV A, R1
 CLR C
 SUBB A, R2
 
 JC ZMIANA 
 LJMP DALEJ
ZMIANA: 
 CPL A 
 INC A
 MOV R5, A
 MOV A, #'-'
 CALL WRITE_DATA
 MOV A, R5
 
DALEJ: 
 MOV B, #10
 DIV AB
 SWAP A
 ADD A, B
 CALL WRITE_HEX
 
 LJMP $
 NOP
